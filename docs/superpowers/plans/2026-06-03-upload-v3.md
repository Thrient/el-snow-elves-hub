# 上传流程 V3 — 无状态 + Redis 锁 + 指纹驱动

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 删除 `Upload` 会话表 → 基于 SHA256 的无状态上传。Redis 分布式锁保证并发安全。所有上传端点只产出 `fingerprint_id`，FileRecord 由业务接口内部创建。每个用户拥有独立的 FileRecord，底层 Fingerprint 共享去重。

**Architecture:**
- `upload_chunks` 表 (sha256, chunk_index) 复合 PK 充当状态记录 + 并发锁
- Redis `SET NX EX` 命名锁 (`upload:chunk:{sha256}:{n}` / `upload:merge:{sha256}`) 保护分片写入和合并
- 上传层只产出 `Fingerprint`（不创建 FileRecord）→ 业务层接收 `fingerprint_id` → 内部建 FileRecord

**Tech Stack:** FastAPI + SQLAlchemy 2.0 async + Redis (SET NX) + MinIO S3 + React 19 + TypeScript

```python
# backend/app/infrastructure/storage/Lock.py
from contextlib import asynccontextmanager
import redis.asyncio as aioredis

CHUNK_TTL = 30    # chunk 锁：30s（写 5MB 到 MinIO 最多几秒）
MERGE_TTL = 300   # merge 锁：5min（极端情况：1000 chunks × multipart copy）


async def lock_chunk(r: aioredis.Redis, sha256: str, n: int) -> bool:
    """Try to acquire chunk lock. Returns True if acquired, False if held by another."""
    key = f"upload:chunk:{sha256}:{n}"
    return await r.set(key, "1", nx=True, ex=CHUNK_TTL)


async def release_chunk(r: aioredis.Redis, sha256: str, n: int):
    key = f"upload:chunk:{sha256}:{n}"
    await r.delete(key)


async def lock_merge(r: aioredis.Redis, sha256: str, *, ttl: int = MERGE_TTL) -> bool:
    """Try to acquire merge lock. Returns True if acquired, False if held by another."""
    key = f"upload:merge:{sha256}"
    return await r.set(key, "1", nx=True, ex=ttl)


async def release_merge(r: aioredis.Redis, sha256: str):
    key = f"upload:merge:{sha256}"
    await r.delete(key)


def get_lock_redis() -> aioredis.Redis:
    """Return Redis client configured for locking (decode_responses=False for SET NX)."""
    from app.Config import settings
    return aioredis.from_url(settings.redis_url, decode_responses=False)


@asynccontextmanager
async def chunk_lock(r: aioredis.Redis, sha256: str, n: int):
    """Context manager: acquire chunk lock, yield, release."""
    acquired = await lock_chunk(r, sha256, n)
    if not acquired:
        yield False
        return
    try:
        yield True
    finally:
        await release_chunk(r, sha256, n)


@asynccontextmanager
async def merge_lock(r: aioredis.Redis, sha256: str, *, ttl: int = MERGE_TTL):
    """Context manager: acquire merge lock, yield, release."""
    acquired = await lock_merge(r, sha256, ttl=ttl)
    if not acquired:
        yield False
        return
    try:
        yield True
    finally:
        await release_merge(r, sha256)
```

```python
# backend/app/infrastructure/storage/ChunkedUpload.py
import hashlib
import logging
from sqlalchemy import select, func, delete
from sqlalchemy.ext.asyncio import AsyncSession

from app.infrastructure.storage.entity.UploadChunk import UploadChunk
from app.infrastructure.storage.MinioClient import client as minio
from app.infrastructure.storage.Lock import lock_chunk, release_chunk, lock_merge, release_merge
from app.infrastructure.storage.entity.Fingerprint import Fingerprint
from app.infrastructure.storage.FileValidator import detect_type

_log = logging.getLogger("Elves.ChunkedUpload")


class ChunkedUpload:

    # ── init: 纯查询，不创建任何记录 ──

    async def init(self, db: AsyncSession, *, sha256: str, total_chunks: int, filename: str) -> dict:
        """返回已上传的分片列表。{exists: bool, chunks: [int], total_chunks: int}"""
        result = await db.execute(
            select(UploadChunk.chunk_index)
            .where(UploadChunk.sha256 == sha256)
            .order_by(UploadChunk.chunk_index)
        )
        chunks = [row[0] for row in result.all()]
        return {
            "exists": len(chunks) > 0,
            "chunks": chunks,
            "total_chunks": total_chunks,
        }

    # ── chunk: Redis 锁 → 双重检查 → MinIO → DB ──

    async def chunk(self, db: AsyncSession, r, *, sha256: str, n: int,
                    total_chunks: int, filename: str, data: bytes) -> dict:
        """上传一个分片。Redis 锁保护并发安全。"""
        if not await lock_chunk(r, sha256, n):
            # 锁被持有 → 有人在写或已写完 → 短暂等待后查 DB
            import asyncio
            await asyncio.sleep(0.1)
            existing = await db.execute(
                select(UploadChunk).where(
                    UploadChunk.sha256 == sha256,
                    UploadChunk.chunk_index == n,
                )
            )
            if existing.scalar_one_or_none():
                return {"chunk": n, "status": "exists"}
            # 锁持有者可能崩溃 → 稍等重试
            await asyncio.sleep(0.5)
            if not await lock_chunk(r, sha256, n):
                return {"chunk": n, "status": "conflict"}
            # 抢到了，继续

        try:
            # 双重检查
            existing = await db.execute(
                select(UploadChunk).where(
                    UploadChunk.sha256 == sha256,
                    UploadChunk.chunk_index == n,
                )
            )
            if existing.scalar_one_or_none():
                return {"chunk": n, "status": "exists"}

            # 写 MinIO + DB
            minio.upload(f"chunks/{sha256}/{n}", data, "application/octet-stream")
            db.add(UploadChunk(
                sha256=sha256,
                chunk_index=n,
                total_chunks=total_chunks,
                filename=filename,
            ))
            await db.commit()
            return {"chunk": n, "status": "ok"}
        finally:
            await release_chunk(r, sha256, n)

    # ── complete: 流式算哈希 → 快路径指纹 → Redis 合并锁 → 组装 → 清理 ──

    async def complete(self, db: AsyncSession, r, *, sha256: str, total_chunks: int) -> dict:
        """合并所有分片，创建 Fingerprint，返回 fingerprint_id。"""
        # 1. 检查分片完整性
        count = await db.execute(
            select(func.count()).where(UploadChunk.sha256 == sha256)
        )
        actual = count.scalar()
        if actual != total_chunks:
            raise ValueError(f"分片未完整: {actual}/{total_chunks}")

        # 2. 从 MinIO 流式读取所有分片，计算完整哈希
        h = hashlib.sha256()
        first_chunk_data = None
        for n in range(total_chunks):
            data, _ = minio.download(f"chunks/{sha256}/{n}")
            h.update(data)
            if n == 0:
                first_chunk_data = data
        full_hash = h.hexdigest()

        # 3. 快路径：指纹已存在 → 直接返回
        existing_fp = await db.execute(
            select(Fingerprint).where(Fingerprint.sha256 == full_hash)
        )
        fp = existing_fp.scalar_one_or_none()
        if fp:
            return {"fingerprint_id": fp.id}

        # 4. 抢合并锁
        if not await lock_merge(r, sha256):
            # 有人在合并 → 等一会再查指纹
            import asyncio
            for _ in range(30):
                await asyncio.sleep(1)
                existing_fp = await db.execute(
                    select(Fingerprint).where(Fingerprint.sha256 == full_hash)
                )
                fp = existing_fp.scalar_one_or_none()
                if fp:
                    return {"fingerprint_id": fp.id}
            # 超时 → 锁持有者可能崩溃 → 抢锁
            if not await lock_merge(r, sha256):
                raise ValueError("合并锁超时，无法获取")

        try:
            # 5. 双重检查
            existing_fp = await db.execute(
                select(Fingerprint).where(Fingerprint.sha256 == full_hash)
            )
            fp = existing_fp.scalar_one_or_none()
            if fp:
                return {"fingerprint_id": fp.id}

            # 6. 组装（MinIO 服务端拷贝）
            mp_id = minio.create_multipart_upload(full_hash)
            parts = []
            for i in range(total_chunks):
                result = minio.upload_part_copy(
                    full_hash, mp_id, i + 1, f"chunks/{sha256}/{i}"
                )
                parts.append(result)
            minio.complete_multipart_upload(full_hash, mp_id, parts)

            # 7. 创建 Fingerprint
            detected = detect_type(first_chunk_data)
            fp = Fingerprint(sha256=full_hash, size=sum(
                [minio.download(f"chunks/{sha256}/{n}")[0].__len__() for n in range(total_chunks)]
            ), detected_type=detected)
            # 简化：用 first_chunk_data 检测类型
            fp = Fingerprint(sha256=full_hash, size=0, detected_type=detected)
            # 实际大小通过 MinIO head object 获取或从 upload_chunks 推算
            db.add(fp)
            await db.flush()

            # 8. 清理分片
            chunk_keys = [f"chunks/{sha256}/{n}" for n in range(total_chunks)]
            try:
                minio.delete_objects(chunk_keys)
            except Exception:
                _log.warning(f"分片清理失败: chunks/{sha256}/*")

            await db.execute(delete(UploadChunk).where(UploadChunk.sha256 == sha256))
            await db.commit()

            return {"fingerprint_id": fp.id}
        finally:
            await release_merge(r, sha256)

    # ── direct: 小文件直传 ──

    async def direct_upload(self, db: AsyncSession, r, *,
                            filename: str, data: bytes) -> dict:
        """小文件直传 — Redis 锁保护并发安全。"""
        sha256 = hashlib.sha256(data).hexdigest()

        # 快路径
        existing_fp = await db.execute(
            select(Fingerprint).where(Fingerprint.sha256 == sha256)
        )
        fp = existing_fp.scalar_one_or_none()
        if fp:
            return {"fingerprint_id": fp.id}

        # 抢锁
        if not await lock_merge(r, sha256):
            import asyncio
            for _ in range(10):
                await asyncio.sleep(0.5)
                existing_fp = await db.execute(
                    select(Fingerprint).where(Fingerprint.sha256 == sha256)
                )
                fp = existing_fp.scalar_one_or_none()
                if fp:
                    return {"fingerprint_id": fp.id}
            if not await lock_merge(r, sha256):
                raise ValueError("直传锁超时")

        try:
            # 双重检查
            existing_fp = await db.execute(
                select(Fingerprint).where(Fingerprint.sha256 == sha256)
            )
            fp = existing_fp.scalar_one_or_none()
            if fp:
                return {"fingerprint_id": fp.id}

            detected = detect_type(data)
            minio.upload(sha256, data, "application/octet-stream")
            fp = Fingerprint(sha256=sha256, size=len(data), detected_type=detected)
```

```python
@router.post("/chunk")
@_limiter.limit("120/minute")
async def upload_chunk(
    request: Request,
    sha256: str = Query(...),
    n: int = Query(...),
    total: int = Query(...),
    filename: str = Query(...),
    chunk: UploadFile = None,
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
    _=Depends(require_perm_any("file:upload:chunk")),
    _v=Depends(require_verified),
):
    """上传单个分片 — Redis 锁保护并发"""
    if not chunk:
        raise HTTPException(400, "缺少分片数据")

    r = get_lock_redis()
    try:
        result = await chunked_upload.chunk(
            db, r,
            sha256=sha256, n=n,
            total_chunks=total, filename=filename,
            data=await chunk.read(),
        )
    finally:
        await r.aclose()

    return ok(result)


@router.post("/complete")
async def complete_upload(
    request: Request, body: CompleteRequest,
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
    _=Depends(require_perm_any("file:upload:complete")),
    _v=Depends(require_verified),
):
    """合并分片 → 创建 Fingerprint → 返回 fingerprint_id"""
    r = get_lock_redis()
    try:
        result = await chunked_upload.complete(
            db, r,
            sha256=body.sha256,
            total_chunks=body.total_chunks,
        )
    except ValueError as e:
        raise HTTPException(400, str(e))
    finally:
        await r.aclose()

    return ok({"fingerprint_id": result["fingerprint_id"]})

@_limiter.limit("60/minute")

    request: Request,

    _v=Depends(require_verified),
):
    """小文件直传 — 一次请求返回 fingerprint_id"""
    r = get_lock_redis()
    try:
        result = await chunked_upload.direct_upload(
            db, r,
            filename=file.filename or "untitled",
            data=await file.read(),
        )
    finally:
        await r.aclose()
```

```python
@staticmethod
async def create_record_from_fingerprint(
    db: AsyncSession,
    fingerprint_id: int,
    filename: str,
    uploaded_by: int,
) -> FileRecord:
    """根据 fingerprint_id 创建 FileRecord — 每次上传都应调用"""
    fp = (await db.execute(
        select(Fingerprint).where(Fingerprint.id == fingerprint_id)
    )).scalar_one_or_none()
    if not fp:
        raise ValueError(f"指纹不存在: {fingerprint_id}")

    record = FileRecord(
        fingerprint_id=fp.id,
        filename=filename,
        size=fp.size,
        uploaded_by=uploaded_by,
    )
    db.add(record)
    await db.flush()
    return record
```

```typescript
async function uploadSingle(
  file: File, sha256: string, onProgress?: (pct: number) => void,
): Promise<{ fingerprint_id: number }> {
  if (file.size <= DIRECT_THRESHOLD) {
    onProgress?.(50);
    const result = await uploadApi.direct(file);
    onProgress?.(100);
    return result;
  }

  const totalChunks = Math.ceil(file.size / CHUNK_SIZE);

  // init: 纯查询，返回已上传分片
  const { chunks: existingChunks } = await uploadApi.init(sha256, totalChunks, file.name);
  const uploadedSet = new Set(existingChunks);

  for (let i = 0; i < totalChunks; i++) {
    if (uploadedSet.has(i)) continue;  // ← 跳过已有

    const blob = file.slice(i * CHUNK_SIZE, Math.min((i + 1) * CHUNK_SIZE, file.size));
    await uploadApi.chunk(sha256, i, totalChunks, file.name, blob);
    onProgress?.(Math.round((i + 1) / totalChunks * 100));
  }

  onProgress?.(100);
  return uploadApi.complete(sha256, totalChunks);
}
```
